            public static Process GetGameProcess(string processName)
            {
                Process pro = null;
                Process[] pros = Process.GetProcessesByName(processName);
                if (pros.Length > 0)
                {
                    pro = pros[0];
                }
                IntPtr m_WindowHandle = IntPtr.Zero;
                m_WindowHandle = pro.MainWindowHandl; 
                return pro;
            }
